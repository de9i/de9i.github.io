void BreadMachine_Update(BreadMachineControl* ctrl) { switch (ctrl->currentState) { case STATE_IDLE: // 等待用户启动 break; case STATE_PREHEAT: // 预热至设定温度 ctrl->currentTemp = DS18B20_ReadTemperature(); if (ctrl->currentTemp >= 350) { // 35 0°C ctrl->currentState = STATE_KNEADING; ctrl->timer = 0; } break; case STATE_KNEADING: // 控制电机揉面（此处简化） if (ctrl->timer >= 900) { // 15分钟 ctrl->currentState = STATE_RISING; ctrl->timer = 0; } break; case STATE_RISING: // 面团发酵 if (ctrl->timer >= 1800) { // 30分钟 ctrl->currentState = STATE_BAKING; ctrl->timer = 0; Heater_SetPower(100); // 全功率加热 } break; case STATE_BAKING: ctrl->currentTemp = DS18B20_ReadTemperature(); if (ctrl->timer >= 1800 || ctrl->currentTemp >= 2000) { // 30分钟或200°C Heater_SetPower(0); ctrl->currentState = STATE_COOLING; ctrl->timer = 0; } break;
